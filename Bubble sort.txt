# Creating a bubble sort function
def bubble_sort(arr):
    # Outer loop for traverse the entire list
    for i in range(0, len(arr) - 1):
        for j in range(len(arr) - 1):
            if (arr[j] > arr[j + 1]):
                temp = arr[j]
                arr[j] = arr[j + 1]
                arr[j + 1] = temp
    return arr
